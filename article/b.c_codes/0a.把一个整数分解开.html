                     数字的分解

最近在翻看C语言的书, 自己也买了一本K&R原版的, K&R的好处在于讲解各种函数的由来和系统接口! 但是中文的C语言则很多是关于一些简单处理, 比如strcpy,strncat,strcat这些函数的实现.... 一时心血来潮, 突然想出了如果给出一个整数, 这个整数有很多位(我并不知道它有多大?), 怎么来把它分解出来? 这个虽然简单, 但毕竟是自己一点点想出来的, 留与此地, 方便以后查阅..... 


#include      
int get_cnum(int x)  
{  
	    int     count = 0;  
	    int     cp = 10;  
	    while (x/cp) {  
           count++;  
           cp *= 10;      
    	}  
    	return count+1;  
}  
int main(void)  
{  
	    int     x = 147258369;  
	    int k = get_cnum(x);  
	    int     i;  
	    int     cp = 1;  
	    int     rev;  
	    for (i = 1; i < k; i++)  
	        cp *= 10;  
	    printf("\n%d--%d\n", get_cnum(x), cp);        
	    while ((x/cp) && (cp != 0)) {  
	        rev = x / cp;  
	        printf("\n%d-\n",rev);  
	        x   %= cp;  
	        cp  /= 10;  
	    }  
	    return 0;  
}
